35def _default_llm(model: str) -> OpenAIResponsesChatGenerator: 

36 """ 

37 A default OpenAI Responses-API generator with the pack's timeout/retry settings. 

38 

39 Reasoning effort is set to "low": enough deliberate thinking for tool sequencing and filter construction, without 

40 compounding latency across the agent's steps (leaving it unset would default to the heavier "medium"). 

41 

42 :param model: The OpenAI model name. 

43 :returns: The generator. 

44 """ 

45 return OpenAIResponsesChatGenerator( 

46 model=model, 

47 timeout=_DEFAULT_TIMEOUT, 

48 max_retries=_DEFAULT_MAX_RETRIES, 

49 generation_kwargs={"reasoning": {"effort": "low"}}, 

50 )

53def create_advanced_rag_agent( 

54 *, 

55 document_store: DocumentStore, 

56 retriever: TextRetriever | Pipeline | None = None, 

57 retrieval_pipeline_input_mapping: dict[str, list[str]] | None = None, 

58 retrieval_pipeline_output_mapping: dict[str, str] | None = None, 

59 llm: ChatGenerator | None = None, 

60 backup_answer_llm: ChatGenerator | None = None, 

61 system_prompt: str | None = None, 

62 max_agent_steps: int = 20, 

63 max_fetched_docs: int = 10, 

64 extra_tools: ToolsType | None = None, 

65 state_schema: dict[str, Any] | None = None, 

66 hooks: dict[HookPoint, list[Hook]] | None = None, 

67 raise_on_tool_invocation_failure: bool = False, 

68 tool_concurrency_limit: int = 4, 

69) -> Agent: 

70 """ 

71 Create the advanced RAG agent. 

72 

73 The agent answers questions from documents it retrieves out of the document store. Instead of guessing which 

74 metadata fields exist, it can inspect the store (fields, values, ranges) and construct a Haystack filter to 

75 narrow its retrieval when metadata helps — plain, unfiltered retrieval remains available when it doesn't. The 

76 answer cites the retrieved documents. 

77 

78 The required `retriever` becomes the `search_documents` tool; `document_store` additionally feeds the three 

79 metadata inspection tools and must implement the metadata introspection methods (`get_metadata_fields_info`, 

80 `get_metadata_field_unique_values`, `get_metadata_field_min_max`). 

81 

82 :param document_store: The document store the metadata inspection tools and the `fetch_documents_by_filter` tool 

83 run against. 

84 :param retriever: What retrieves for the `search_documents` tool (required). Either a standalone retriever 

85 component following the `TextRetriever` protocol, i.e. its `run` method accepts `query` and `filters` 

86 (e.g. `InMemoryBM25Retriever`, or an embedding retriever wrapped in `TextEmbeddingRetriever`), or a custom 

87 retrieval `Pipeline` (e.g. embedder -> retriever, or hybrid retrieval) — a pipeline additionally requires 

88 `retrieval_pipeline_input_mapping`. It should retrieve by relevance scoring (keyword or embedding-based) — 

89 direct, unscored fetching is already covered by the built-in `fetch_documents_by_filter` tool. 

90 :param retrieval_pipeline_input_mapping: Required when `retriever` is a `Pipeline`: maps the tool inputs to 

91 pipeline input sockets; must have exactly the keys "query" and "filters", 

92 e.g. `{"query": ["embedder.text"], "filters": ["retriever.filters"]}`. 

93 :param retrieval_pipeline_output_mapping: Optional when `retriever` is a `Pipeline`: maps pipeline output sockets 

94 to tool outputs, e.g. `{"retriever.documents": "documents"}`. 

95 :param llm: LLM that drives the agent loop. Defaults to `OpenAIResponsesChatGenerator("gpt-5.4")` with low 

96 reasoning effort. 

97 :param backup_answer_llm: LLM the built-in `BackupAnswerHook` uses to write a best-effort answer when the run 

98 is cut off by `max_agent_steps`. Defaults to a separate `OpenAIResponsesChatGenerator("gpt-5.4")` with low 

99 reasoning effort. 

100 :param system_prompt: Overrides the pre-made system prompt. 

101 :param max_agent_steps: Maximum steps for the agent loop. If the loop is cut off by this limit before writing an 

102 answer, an `after_run` hook (`BackupAnswerHook`) makes one extra LLM call to produce a best-effort answer from 

103 the evidence gathered so far, so `last_message` always carries a text answer. 

104 :param max_fetched_docs: Maximum number of documents `fetch_documents_by_filter` shows per fetch. A filter fetch is 

105 not bounded by a retriever's `top_k`, so this caps the tool result instead; the scored `search_documents` tool 

106 is bounded by the `top_k` configured on your retrieval components. 

107 :param extra_tools: Additional tools (or toolsets) for the agent, appended after the built-in document-store 

108 toolset and the retrieval tool. 

109 :param state_schema: Additional entries merged into the agent's state schema. The built-in `documents` entry 

110 (the accumulated retrieved documents) always takes precedence. 

111 :param hooks: Additional hooks per hook point, merged with the built-in hooks. For `after_run`, the built-in 

112 backup-answer hook runs first, so custom hooks see the final answer. 

113 :param raise_on_tool_invocation_failure: If True, a failing tool call raises instead of being returned to the LLM 

114 as an error message it can recover from (the default). 

115 :param tool_concurrency_limit: Maximum number of tool calls executed in parallel within one agent step. 

116 :returns: The advanced RAG `Agent`. Call it with the question as a user message, 

117 `agent.run(messages=[ChatMessage.from_user(question)])`; the answer is in `last_message` (a `ChatMessage`) and 

118 `documents` carries every document the agent retrieved during the run (deduplicated by id, in first-retrieved 

119 order) — the answer cites them by the first 8 characters of their id, e.g. `[doc a1b2c3d4]`. The standard Agent 

120 outputs `messages`, `step_count`, `token_usage` and `tool_call_counts` are also returned. 

121 """ 

122 if system_prompt is None: 

123 # Only the default system prompt requires the `{% now %}` Jinja tag (and thus `arrow`). 

124 arrow_import.check() 

125 

126 if retriever is None: 

127 msg = "`retriever` is required: pass a retriever component or a retrieval `Pipeline`." 

128 raise ValueError(msg) 

129 

130 retrieval_tool: Tool 

131 if isinstance(retriever, Pipeline): 

132 if retrieval_pipeline_input_mapping is None: 

133 msg = "`retrieval_pipeline_input_mapping` is required when `retriever` is a `Pipeline`." 

134 raise ValueError(msg) 

135 retrieval_tool = _make_retrieval_pipeline_tool( 

136 pipeline=retriever, 

137 input_mapping=retrieval_pipeline_input_mapping, 

138 output_mapping=retrieval_pipeline_output_mapping, 

139 ) 

140 else: 

141 if retrieval_pipeline_input_mapping is not None or retrieval_pipeline_output_mapping is not None: 

142 msg = "The `retrieval_pipeline_*` arguments are only valid when `retriever` is a `Pipeline`." 

143 raise ValueError(msg) 

144 retrieval_tool = _make_retriever_tool(retriever=retriever) 

145 

146 llm = llm or _default_llm("gpt-5.4") 

147 backup_answer_llm = backup_answer_llm or _default_llm("gpt-5.4") 

148 

149 tools: list[Tool | Toolset] = [ 

150 DocumentStoreToolset(document_store, max_fetched_docs=max_fetched_docs), 

151 retrieval_tool, 

152 ] 

153 if extra_tools is not None: 

154 tools.extend([extra_tools] if isinstance(extra_tools, Toolset) else list(extra_tools)) 

155 

156 merged_hooks: dict[HookPoint, list[Hook]] = {"after_run": [BackupAnswerHook(chat_generator=backup_answer_llm)]} 

157 for hook_point, point_hooks in (hooks or {}).items(): 

158 merged_hooks.setdefault(hook_point, []).extend(point_hooks) 

159 

160 return Agent( 

161 chat_generator=llm, 

162 system_prompt=system_prompt or prompts.SYSTEM_TEMPLATE, 

163 tools=tools, 

164 exit_conditions=["text"], 

165 max_agent_steps=max_agent_steps, 

166 state_schema={**(state_schema or {}), "documents": {"type": list[Document]}}, 

167 hooks=merged_hooks, 

168 raise_on_tool_invocation_failure=raise_on_tool_invocation_failure, 

169 tool_concurrency_limit=tool_concurrency_limit, 

170 )
