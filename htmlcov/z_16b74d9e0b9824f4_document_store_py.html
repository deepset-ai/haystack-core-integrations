643 # Try to parse as JSON to get the actual type 

644 try: 

645 parsed_value = json.loads(value) 

646 # Use JSON string for deduplication to handle unhashable types 

647 value_key = json.dumps(parsed_value, sort_keys=True) 

648 if value_key not in seen: 

649 seen.add(value_key) 

650 values.append(parsed_value) 

651 except (json.JSONDecodeError, TypeError): 

652 # If it's not valid JSON, use the string value 

653 if value not in seen: 

654 seen.add(value) 

655 values.append(value) 

656 

657 return values

659 def get_metadata_field_min_max(self, field: str) -> dict[str, Any]: 

660 """ 

661 Get the minimum and maximum values for a numeric metadata field. 

662 

663 :param field: The metadata field name (can include 'meta.' prefix) 

664 :return: Dictionary with 'min' and 'max' keys 

665 """ 

666 # Strip 'meta.' prefix if present 

667 field_name = field.removeprefix("meta.") 

668 

669 conn = self._get_connection() 

670 

671 with conn.cursor() as cur: 

672 # Try to get min/max treating the field as numeric 

673 # Use RETURNING VARCHAR to explicitly specify the return type for JSON_VALUE 

674 sql = ( 

675 f"SELECT " 

676 f"MIN(CAST(JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) AS DOUBLE)), " 

677 f"MAX(CAST(JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) AS DOUBLE)) " 

678 f"FROM {self.table_name} " 

679 f"WHERE JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) IS NOT NULL" 

680 ) 

681 try: 

682 cur.execute(sql) 

683 row = cur.fetchone() 

684 if row and row[0] is not None: 

685 return {"min": row[0], "max": row[1]} 

686 except Exception: 

687 # If numeric cast fails, try lexicographic comparison 

688 sql = ( 

689 f"SELECT " 

690 f"MIN(JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000))), " 

691 f"MAX(JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000))) " 

692 f"FROM {self.table_name} " 

693 f"WHERE JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) IS NOT NULL" 

694 ) 

695 try: 

696 cur.execute(sql) 

697 row = cur.fetchone() 

698 if row and row[0] is not None: 

699 return {"min": row[0], "max": row[1]} 

700 except Exception as e: 

701 msg = f"Failed to get min/max for field '{field}': {e}" 

702 raise RuntimeError(msg) from e 

703

706 def get_metadata_fields_info(self) -> dict[str, dict[str, Any]]: 

707 """ 

708 Get information about all metadata fields including their types. 

709 

710 :return: Dictionary mapping field names to their type information 

711 """ 

712 conn = self._get_connection() 

713 

714 with conn.cursor() as cur: 

715 try: 

716 # Get all metadata from documents 

717 sql = f"SELECT SYSTOOLS.BSON2JSON(meta) AS meta FROM {self.table_name} WHERE meta IS NOT NULL" 

718 cur.execute(sql) 

719 rows = cur.fetchall() 

720 except Exception as e: 

721 msg = f"Failed to get metadata fields info: {e}" 

722 raise RuntimeError(msg) from e 

723 

724 # Analyze the metadata to infer field types 

725 fields_info: dict[str, dict[str, Any]] = {} 

726 

727 for row in rows: 

728 meta_json = row[0] 

729 if not meta_json: 

730 continue 

731 

732 try: 

733 meta = json.loads(meta_json) 

734 if not isinstance(meta, dict): 

735 continue 

736 

737 for field_name, field_value in meta.items(): 

738 if field_name not in fields_info: 

739 if field_value is not None: 

740 field_type = self._infer_field_type(field_value) 

741 fields_info[field_name] = {"type": field_type} 

742 else: 

743 fields_info[field_name] = {"type": "text"} 

744 except (json.JSONDecodeError, TypeError): 

745 continue 

746 

747 return fields_info

749 def count_unique_metadata_by_filter( 

750 self, filters: dict[str, Any] | None = None, metadata_fields: list[str] | None = None 

751 ) -> dict[str, int]: 

752 """ 

753 Count unique values for specified metadata fields, optionally filtered. 

754 

755 :param filters: Optional filters to apply before counting 

756 :param metadata_fields: List of metadata field names to count unique values for 

757 :return: Dictionary mapping field names to their unique value counts 

758 """ 

759 if not metadata_fields: 

760 return {} 

761 

762 conn = self._get_connection() 

763 where_clause, params = self._build_where_clause(filters) if filters else ("", []) 

764 

765 result = {} 

766 for field in metadata_fields: 

767 # Strip 'meta.' prefix if present 

768 field_name = field.removeprefix("meta.") 

769 

770 # Count distinct values for this field 

771 # We need to fetch all values and deduplicate in Python due to DB2's JSON handling 

772 with conn.cursor() as cur: 

773 try: 

774 sql = ( 

775 f"SELECT JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) " 

776 f"FROM {self.table_name} " 

777 ) 

778 if where_clause: 

779 sql += where_clause + " AND " 

780 else: 

781 sql += "WHERE " 

782 sql += f"JSON_VALUE(SYSTOOLS.BSON2JSON(meta), '$.{field_name}' RETURNING VARCHAR(1000)) IS NOT NULL" 

783 

784 cur.execute(sql, params) 

785 rows = cur.fetchall() 

786 except Exception as e: 

787 msg = f"Failed to count unique metadata for field '{field}': {e}" 

788 raise RuntimeError(msg) from e 

789 

790 # Deduplicate values 

791 unique_values = set() 

792 for row in rows: 

793 value = row[0] 

794 if value is not None: 

795 # Try to parse as JSON to handle different types consistently 

796 try: 

797 parsed_value = json.loads(value) 

798 # Use JSON string for deduplication to handle unhashable types 

799 value_key = json.dumps(parsed_value, sort_keys=True) 

800 unique_values.add(value_key) 

801 except (json.JSONDecodeError, TypeError): 

802 # If it's not valid JSON, use the string value 

803 unique_values.add(value) 

804 

805 result[field] = len(unique_values) 

806 

807 return result

861 def _embedding_retrieval( 

862 self, 

863 query_embedding: list[float], 

864 *, 

865 filters: dict[str, Any] | None = None, 

866 top_k: int = 10, 

867 ) -> list[Document]: 

868 """ 

869 Retrieve documents by embedding similarity. 

870 

871 :param query_embedding: Query embedding vector 

872 :param filters: Optional filters to apply 

873 :param top_k: Number of documents to retrieve 

874 :return: List of documents with similarity scores 

875 :raises ValueError: If query_embedding is invalid 

876 :raises TypeError: If query_embedding has invalid type 

877 """ 

878 # Validate query embedding 

879 self._validate_embedding(query_embedding, allow_none=False) 

880 

881 conn = self._get_connection() 

882 embedding_str = f"{query_embedding}" 

883 where_clause, filter_params = self._build_where_clause(filters) if filters else ("", []) 

884 

885 # Add condition to exclude NULL embeddings 

886 null_check = "embedding IS NOT NULL" 

887 if where_clause: 

888 where_clause = f"{where_clause} AND {null_check}" 

889 else: 

890 where_clause = f"WHERE {null_check}" 

891 

892 sql = ( 

893 f"SELECT id, content, SYSTOOLS.BSON2JSON(meta) AS meta, embedding, " 

894 f"VECTOR_DISTANCE(embedding, VECTOR(CAST(? AS CLOB(100000)), {self.embedding_dim}, FLOAT32), " 

895 f"{self.distance_metric}) AS score " 

896 f"FROM {self.table_name} " 

897 f"{where_clause} " 

898 f"ORDER BY score ASC FETCH FIRST ? ROWS ONLY" 

899 ) 

900 params: list[Any] = [embedding_str, *filter_params, top_k] 

901 

902 with conn.cursor() as cur: 

903 cur.execute(sql, params) 

904 try: 

905 rows = cur.fetchall() 

906 except BaseException as e: 

907 # If we get a division by zero error (SQL0801N), it means there are zero vectors 

908 # In this case, we'll return empty results or filter them out 

909 error_msg = str(e) 

910 # Check both the error message and the __cause__ attribute 

911 cause_msg = str(e.__cause__) if hasattr(e, "__cause__") and e.__cause__ else "" 

912 if ( 

913 "SQL0801N" in error_msg 

914 or "Division by zero" in error_msg 

915 or "SQL0801N" in cause_msg 

916 or "Division by zero" in cause_msg 

917 ): 

918 # For COSINE metric with zero vectors, return empty results 

919 # This is an edge case that shouldn't happen in production 

920 rows = [] 

921 else: 

922 raise 

923 

924 documents = [] 

925 for row in rows: 

926 doc_id, content, meta_json, embedding, score = row 

927 meta = json.loads(meta_json) if meta_json else {} 

928 embedding_list = _parse_embedding(embedding) 

929 doc = Document( 

930 id=doc_id, 

931 content=content, 

932 meta=meta, 

933 embedding=embedding_list, 

934 score=float(score), 

935 ) 

936 documents.append(doc) 

937 

938 return documents
