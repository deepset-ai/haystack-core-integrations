12def url_validation(api_url: str) -> str: 

13 """ 

14 Validate and normalize an API URL. 

15 

16 :param api_url: 

17 The API URL to validate and normalize. 

18 :returns: 

19 A normalized version of the API URL with '/v1' path appended, if needed. 

20 :raises ValueError: 

21 If the base URL path is not recognized or does not match expected format. 

22 """ 

23 if api_url is not None: 

24 parsed = urlparse(api_url) 

25 

26 # Ensure scheme and netloc (domain name) are present 

27 if not (parsed.scheme and parsed.netloc): 

28 expected_format = "Expected format is: http://host:port" 

29 msg = f"Invalid api_url format. {expected_format} Got: {api_url}" 

30 raise ValueError(msg) 

31 

32 normalized_path = parsed.path.rstrip("/") 

33 if not normalized_path.endswith("/v1"): 

34 warnings.warn(f"{api_url} does not end in /v1, you may have inference and listing issues", stacklevel=2) 

35 normalized_path += "/v1" 

36 

37 api_url = urlunparse((parsed.scheme, parsed.netloc, normalized_path, None, None, None)) 

38 return api_url

49def lookup_model(name: str) -> Model | None: 

50 """ 

51 Lookup a model by name, using only the table of known models. 

52 

53 The name is either: 

54 - directly in the table 

55 - an alias in the table 

56 - not found (None) 

57 Callers can check to see if the name was an alias by 

58 comparing the result's id field to the name they provided. 

59 """ 

60 if not (model := MODEL_TABLE.get(name)): 

61 for mdl in MODEL_TABLE.values(): 

62 if mdl.aliases and name in mdl.aliases: 

63 model = mdl 

64 break 

65 return model

68def determine_model(name: str) -> Model | None: 

69 """ 

70 Determine the model to use based on a name, using only the table of known models. 

71 

72 Raise a warning if the model is found to be 

73 an alias of a known model. 

74 

75 If the model is not found, return None. 

76 """ 

77 if model := lookup_model(name): 

78 # all aliases are deprecated 

79 if model.id != name: 

80 warn_msg = f"Model {name} is deprecated. Using {model.id} instead." 

81 warnings.warn(warn_msg, UserWarning, stacklevel=1) 

82 return model

85def validate_hosted_model( 

86 model_name: str, 

87 client: Client | None = None, 

88) -> Model: 

89 """ 

90 Checks if a given model is compatible with given client. 

91 

92 :param model_name: The name of the model. 

93 :param client: client name enum, e.g. Client.NVIDIA_GENERATOR, Client.NVIDIA_RANKER, 

94 Client.NVIDIA_TEXT_EMBEDDER, Client.NVIDIA_DOCUMENT_EMBEDDER 

95 

96 :raises ValueError: If the model is incompatible with the client or if the model is unknown. 

97 

98 :returns: The model if it is compatible with the client. 

99 """ 

100 supported = { 

101 Client.NVIDIA_GENERATOR: ("chat",), 

102 Client.NVIDIA_TEXT_EMBEDDER: ("embedding",), 

103 Client.NVIDIA_DOCUMENT_EMBEDDER: ("embedding",), 

104 Client.NVIDIA_RANKER: ("ranking",), 

105 } 

106 

107 if model := determine_model(model_name): 

108 err_msg = f"Model {model.id} is incompatible with client {client}. Please check `{client}.available_models`." 

109 

110 if client and model.client and model.client != client: 

111 raise ValueError(err_msg) 

112 elif client and not model.client and model.model_type not in supported[client]: 

113 raise ValueError(err_msg) 

114 elif not client: 

115 warn_msg = f"Unable to determine validity of {model.id}" 

116 warnings.warn(warn_msg, stacklevel=1) 

117 

118 else: 

119 err_msg = f"Model {model_name} is unknown, check `available_models`" 

120 raise ValueError(err_msg) 

121 return model
