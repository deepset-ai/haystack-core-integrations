5# The two agents and the composition that wires them into one deep research Agent. 

6# 

7# - researcher: a reusable sub-agent that researches ONE sub-question in its own isolated context and returns a 

8# compressed, cited summary. 

9# - deep research agent: 

10# - before starting the Agent loop, turns the user query into a research brief using the Scope Hook. 

11# - delegates sub-questions to the researcher and accumulates the returned summaries in shared `State`. 

12# - after the Agent loop, turns the brief plus collected notes into the final report using the Write Hook. 

13

36def _default_llm(model: str) -> OpenAIResponsesChatGenerator: 

37 """A default OpenAI generator for a stage, with the pack's timeout/retry settings.""" 

38 return OpenAIResponsesChatGenerator(model=model, timeout=_DEFAULT_TIMEOUT, max_retries=_DEFAULT_MAX_RETRIES)

41def _collect_note(current: list[str] | None, message: ChatMessage) -> list[str]: 

42 """State handler: append a researcher's final summary (text) to `notes`.""" 

43 current = current or [] 

44 note = getattr(message, "text", None) # message is the researcher's last ChatMessage 

45 if not note: 

46 logger.info("sub-researcher returned an empty note (likely hit its step cap); skipping") 

47 return current 

48 logger.info( 

49 "note collected (#{index}, {length} chars):\n{note}", index=len(current) + 1, length=len(note), note=note 

50 ) 

51 return [*current, note]

54def _make_researcher_agent( 

55 *, 

56 researcher_llm: ChatGenerator, 

57 summarizer_llm: ChatGenerator, 

58 max_researcher_steps: int, 

59 max_search_results: int, 

60 max_content_length: int, 

61) -> Agent: 

62 """ 

63 The reusable sub-researcher. 

64 

65 :param researcher_llm: LLM that drives the search/read/think loop and writes the summary. 

66 :param summarizer_llm: LLM used inside the `read_url` tool to summarize a page toward the question. 

67 :param max_researcher_steps: Cap on the researcher's agent loop (search/read/think + final summary). 

68 :param max_search_results: Results returned per `web_search` call. 

69 :param max_content_length: Raw page chars fed to the summarizer (pre-summarization cap). 

70 """ 

71 tavily_import.check() 

72 return Agent( 

73 chat_generator=researcher_llm, 

74 system_prompt=prompts.RESEARCHER_TEMPLATE, 

75 tools=[ 

76 TavilyWebSearchTool(top_k=max_search_results), 

77 make_read_url_tool(summarizer_llm=summarizer_llm, max_content_length=max_content_length), 

78 think_tool, 

79 ], 

80 exit_conditions=["text"], # exits when it writes its summary 

81 max_agent_steps=max_researcher_steps, 

82 )

106def create_deep_research_agent( 

107 *, 

108 scope_llm: ChatGenerator | None = None, 

109 orchestrator_llm: ChatGenerator | None = None, 

110 researcher_llm: ChatGenerator | None = None, 

111 summarizer_llm: ChatGenerator | None = None, 

112 writer_llm: ChatGenerator | None = None, 

113 max_subtopics: int = 5, 

114 max_concurrent_researchers: int = 5, 

115 max_orchestrator_steps: int = 8, 

116 max_researcher_steps: int = 20, 

117 max_search_results: int = 10, 

118 max_content_length: int = 50_000, 

119) -> Agent: 

120 """ 

121 Create the deep research agent. 

122 

123 :param scope_llm: LLM that rewrites the user query into a focused research brief. 

124 Defaults to `OpenAIResponsesChatGenerator("gpt-5.4")`. 

125 :param orchestrator_llm: LLM that plans the investigation and delegates the sub-questions. 

126 Defaults to `OpenAIResponsesChatGenerator("gpt-5.4")`. 

127 :param researcher_llm: LLM that drives each sub-researcher's search/read/think loop. 

128 Defaults to `OpenAIResponsesChatGenerator("gpt-5.4-mini")`. 

129 :param summarizer_llm: LLM used inside the `read_url` tool to summarize a fetched page toward 

130 the question. Defaults to `OpenAIResponsesChatGenerator("gpt-5.4-mini")`. 

131 :param writer_llm: LLM that turns the brief plus collected notes into the final report. 

132 Defaults to `OpenAIResponsesChatGenerator("gpt-5.4")`. 

133 :param max_subtopics: Maximum number of sub-questions the orchestrator may delegate (breadth). 

134 :param max_concurrent_researchers: Maximum number of sub-researchers that run at the same time. 

135 :param max_orchestrator_steps: Maximum steps for the orchestrator's agent loop (reflect -> delegate rounds). 

136 :param max_researcher_steps: Maximum steps for each sub-researcher's agent loop. 

137 :param max_search_results: Number of results returned per `web_search` call. 

138 :param max_content_length: Maximum raw page characters fed to the summarizer, before summarization. 

139 :returns: The deep research `Agent`. Call it with the question as a user message, 

140 `agent.run(messages=[ChatMessage.from_user(question)])`; it returns a dict whose main output is 

141 `report` (the final markdown report, a `str`). The dict also carries the intermediate `brief` 

142 (`str`) and `notes` (`list[str]`), plus the standard Agent outputs `messages`, `last_message`, 

143 `step_count`, `token_usage` and `tool_call_counts`. 

144 """ 

145 scope_llm = scope_llm or _default_llm("gpt-5.4") 

146 orchestrator_llm = orchestrator_llm or _default_llm("gpt-5.4") 

147 researcher_llm = researcher_llm or _default_llm("gpt-5.4-mini") 

148 summarizer_llm = summarizer_llm or _default_llm("gpt-5.4-mini") 

149 writer_llm = writer_llm or _default_llm("gpt-5.4") 

150 

151 researcher = _make_researcher_agent( 

152 researcher_llm=researcher_llm, 

153 summarizer_llm=summarizer_llm, 

154 max_researcher_steps=max_researcher_steps, 

155 max_search_results=max_search_results, 

156 max_content_length=max_content_length, 

157 ) 

158 

159 research_subtopic = ComponentTool( 

160 component=researcher, 

161 name="research_subtopic", 

162 description=( 

163 "Research ONE focused sub-question in an isolated context. " 

164 "Pass the sub-question as a user message. " 

165 "Returns a compressed, cited summary of the findings." 

166 ), 

167 parameters=_RESEARCH_SUBTOPIC_PARAMS, 

168 outputs_to_string={"source": "last_message"}, 

169 outputs_to_state={"notes": {"source": "last_message", "handler": _collect_note}}, 

170 ) 

171 

172 scope = ScopeHook( 

173 generator=scope_llm, 

174 prompt_builder=ChatPromptBuilder(template=prompts.SCOPE_TEMPLATE, required_variables=["query"]), 

175 ) 

176 write = WriteHook( 

177 generator=writer_llm, 

178 prompt_builder=ChatPromptBuilder(template=prompts.WRITER_TEMPLATE, required_variables=["replies", "notes"]), 

179 ) 

180 

181 system_prompt = prompts.ORCHESTRATOR_TEMPLATE.replace("{{ max_subtopics }}", str(max_subtopics)) 

182 

183 return Agent( 

184 chat_generator=orchestrator_llm, 

185 system_prompt=system_prompt, 

186 tools=[research_subtopic, think_tool], 

187 exit_conditions=["text"], 

188 max_agent_steps=max_orchestrator_steps, 

189 tool_concurrency_limit=max_concurrent_researchers, 

190 state_schema={"notes": {"type": list}, "brief": {"type": str}, "report": {"type": str}}, 

191 hooks={"before_run": [scope], "after_run": [write]}, 

192 )
