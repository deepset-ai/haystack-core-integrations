13@component 

14class WatsonxTextEmbedder: 

15 """ 

16 Embeds strings using IBM watsonx.ai foundation models. 

17 

18 You can use it to embed user query and send it to an embedding Retriever. 

19 

20 ### Usage example 

21 

22 ```python 

23 from haystack_integrations.components.embedders.watsonx.text_embedder import WatsonxTextEmbedder 

24 

25 text_to_embed = "I love pizza!" 

26 

27 text_embedder = WatsonxTextEmbedder( 

28 model="ibm/slate-30m-english-rtrvr-v2", 

29 api_key=Secret.from_env_var("WATSONX_API_KEY"), 

30 api_base_url="https://us-south.ml.cloud.ibm.com", 

31 project_id=Secret.from_env_var("WATSONX_PROJECT_ID"), 

32 ) 

33 

34 print(text_embedder.run(text_to_embed)) 

35 

36 # {'embedding': [0.017020374536514282, -0.023255806416273117, ...], 

37 # 'meta': {'model': 'ibm/slate-30m-english-rtrvr-v2', 

38 # 'truncated_input_tokens': 3}} 

39 ``` 

40 """

42 def __init__( 

43 self, 

44 *, 

45 model: str = "ibm/slate-30m-english-rtrvr-v2", 

46 api_key: Secret = Secret.from_env_var("WATSONX_API_KEY"), # noqa: B008 

47 api_base_url: str = "https://us-south.ml.cloud.ibm.com", 

48 project_id: Secret = Secret.from_env_var("WATSONX_PROJECT_ID"), # noqa: B008 

49 truncate_input_tokens: int | None = None, 

50 prefix: str = "", 

51 suffix: str = "", 

52 timeout: float | None = None, 

53 max_retries: int | None = None, 

54 ) -> None: 

55 """ 

56 Creates an WatsonxTextEmbedder component. 

57 

58 :param model: 

59 The name of the IBM watsonx model to use for calculating embeddings. 

60 Default is "ibm/slate-30m-english-rtrvr-v2". 

61 :param api_key: 

62 The WATSONX API key. Can be set via environment variable WATSONX_API_KEY. 

63 :param api_base_url: 

64 The WATSONX URL for the watsonx.ai service. 

65 Default is "https://us-south.ml.cloud.ibm.com". 

66 :param project_id: 

67 The ID of the Watson Studio project. 

68 Can be set via environment variable WATSONX_PROJECT_ID. 

69 :param truncate_input_tokens: 

70 Maximum number of tokens to use from the input text. 

71 If set to `None` (or not provided), the full input text is used, up to the model's maximum token limit. 

72 :param prefix: 

73 A string to add at the beginning of each text to embed. 

74 :param suffix: 

75 A string to add at the end of each text to embed. 

76 :param timeout: 

77 Timeout for API requests in seconds. 

78 :param max_retries: 

79 Maximum number of retries for API requests. 

80 """ 

81 

82 self.model = model 

83 self.api_key = api_key 

84 self.api_base_url = api_base_url 

85 self.project_id = project_id 

86 self.truncate_input_tokens = truncate_input_tokens 

87 self.prefix = prefix 

88 self.suffix = suffix 

89 self.timeout = timeout 

90 self.max_retries = max_retries 

91 

92 # Initialize the embeddings client 

93 credentials = Credentials(api_key=api_key.resolve_value(), url=api_base_url) 

94 

95 params = {} 

96 if truncate_input_tokens is not None: 

97 params["truncate_input_tokens"] = truncate_input_tokens 

98 

99 self.embedder = Embeddings( 

100 model_id=model, 

101 credentials=credentials, 

102 project_id=project_id.resolve_value(), 

103 params=params if params else None, 

104 max_retries=max_retries, 

105 )

146 def _prepare_input(self, text: str) -> str: 

147 if not isinstance(text, str): 

148 msg = ( 

149 "WatsonxTextEmbedder expects a string as an input. " 

150 "In case you want to embed a list of Documents, please use the WatsonxDocumentEmbedder." 

151 ) 

152 raise TypeError(msg) 

153 return self.prefix + text + self.suffix 

154 

155 @component.output_types(embedding=list[float], meta=dict[str, Any]) 

156 def run(self, text: str) -> dict[str, list[float] | dict[str, Any]]: 

157 """ 

158 Embeds a single string. 

159 

160 :param text: Text to embed. 

161 :returns: A dictionary with: 

162 - 'embedding': The embedding of the input text 

163 - 'meta': Information about the model usage 

164 """ 

165 text_to_embed = self._prepare_input(text=text) 

166 embedding = self.embedder.embed_query(text_to_embed) 

167 return { 

168 "embedding": embedding, 

169 "meta": {"model": self.model, "truncated_input_tokens": self.truncate_input_tokens}, 

170 }
